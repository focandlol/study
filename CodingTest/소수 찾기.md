import java.util.*;
class Solution {
    Set<Integer> set = new HashSet<>();
    boolean[] sosu = new boolean[9999999];
    
    public int solution(String numbers) {
        String[] split = numbers.split("");
        sosu[0] = true; sosu[1] = true;
        checkSosu();
        back(split, new StringBuilder(), new boolean[split.length], 0);
        
        return set.size();
    }
    
    public void back(String[] arr, StringBuilder sb, boolean[] visited, int depth){
        if(depth == arr.length){
            return;
        }
        
        for(int i=0; i<arr.length; i++){
            if(visited[i]) continue;
            
            StringBuilder bu = new StringBuilder();
            bu.append(sb).append(arr[i]);
            
            if(!sosu[Integer.parseInt(bu.toString())]){
                set.add(Integer.parseInt(bu.toString()));
            }
            
            visited[i] = true;
            
            back(arr, bu, visited, depth + 1);
            
            visited[i] = false;
        }
    }
    
    public void checkSosu(){
        for(int i=2; i<=Math.sqrt(sosu.length); i++){
            if(sosu[i]) continue;
            
            for(int j=i*i; j<sosu.length; j+=i){
                sosu[j] = true;
            }
        }
    }
}
