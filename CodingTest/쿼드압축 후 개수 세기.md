class Solution {
    static int[] dap = new int[2];
    public int[] solution(int[][] arr) {
        find(0,0,arr.length, arr);
        
        return dap;
    }
    
    public void find(int x,int y, int size, int[][] arr){
        if(check(x,y,size,arr)){
            dap[arr[x][y]]++;
            return;
        }
        
        find(x,y,size/2,arr);
        find(x, y+size/2 , size/2, arr);
        find(x+size/2, y, size/2, arr);
        find(x + size/2, y + size / 2, size / 2, arr);
    }
    
    public boolean check(int x, int y, int size, int[][] arr){
        int first = arr[x][y];
        
        for(int i=x; i<x+size; i++){
            for(int j=y; j<y+size; j++){
                if(first != arr[i][j]) return false;
            }
        }
        
        return true;
    }
}
