import java.util.*;
class Solution {
    public String[] solution(int[][] line) {
        Set<P> set = new HashSet<>();
        long minx = Long.MAX_VALUE;
        long maxx = Long.MIN_VALUE;
        long miny = Long.MAX_VALUE;
        long maxy = Long.MIN_VALUE;
        System.out.println("maxy: " + maxy + ", miny: " + miny);
        for(int i=0; i<line.length; i++){
            long a = line[i][0];
            long b = line[i][1];
            long e = line[i][2];
            for(int j=i+1; j<line.length; j++){
                long c = line[j][0];
                long d = line[j][1];
                long f = line[j][2];
                
                long adbc = a*d - b*c;
                if(adbc == 0) continue;
                
                long bfed = b*f - e*d;
                if(bfed % adbc != 0) continue;
                
                long ecaf = e*c - a*f;
                if(ecaf % adbc != 0) continue;
                
                long x = bfed/adbc;
                long y = ecaf/adbc;
                
                set.add(new P(x, y));
                
                minx = Math.min(minx, x);
                maxx = Math.max(maxx, x);
                miny = Math.min(miny, y);
                maxy = Math.max(maxy, y);
                
            }
        }
        System.out.println("maxy: " + maxy + ", miny: " + miny);
        long height = maxy - miny + 1;
        long width = maxx - minx + 1;
        System.out.println("height : " + height);
        String[] dap = new String[(int)height];
        
        StringBuilder sb = new StringBuilder();
        for(int i=0; i<width; i++){
            sb.append(".");
        }
        Arrays.fill(dap,sb.toString());
        
        for(P p : set){
            long x = p.x - minx;
            long y = maxy - p.y;
            
            dap[(int)y] = dap[(int)y].substring(0, (int)x) + "*" + dap[(int)y].substring((int)x+1);
        }
        return dap;
    }
    
    class P{
        long x;
        long y;
        public P(long x,long y){
            this.x = x;
            this.y = y;
        }
        
        @Override
        public boolean equals(Object o){
            if(this == o) return true;
            if(!(o instanceof P)) return false;
            P p = (P) o;
            return this.x == p.x && this.y == p.y;
        }
        
        @Override
        public int hashCode() {
        return Objects.hash(x, y);
        }
    }
}
