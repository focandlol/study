class Solution {
    int sum = 0;
    int count = 0;
    public int solution(int[] numbers, int target) {
        for(int num : numbers){
            sum += num;
        }
        
        if(sum == target) return 1;
        
        for(int i=1; i<numbers.length; i++){
            back(i, numbers, 0, 0, 0,target);
        }
        
        return count;
    }
    
    void back(int size, int[] numbers, int start, int depth, int gap, int target){
        if(depth == size){
            if(sum - gap * 2 == target){
                count++;
            }
            return;
        }
        
        for(int i=start; i<numbers.length; i++){
            back(size, numbers, i + 1, depth + 1, gap + numbers[i], target);
        }
    }
}
