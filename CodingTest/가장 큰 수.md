import java.util.stream.*;
import java.util.*;
class Solution {
    public String solution(int[] numbers) {
        String dap = Arrays.stream(numbers)
            .mapToObj(String::valueOf)
            .sorted((a,b) -> (b+a).compareTo(a+b))
            .collect(Collectors.joining());
        
        return dap.startsWith("0") ? "0" : dap;
    }
}
