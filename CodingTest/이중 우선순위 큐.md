import java.util.*;
class Solution {
    public int[] solution(String[] operations) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        
        for(String s : operations){
            String[] split = s.split(" ");
            
            if("I".equals(split[0])){
                map.put(Integer.parseInt(split[1]), map.getOrDefault(Integer.parseInt(split[1]), 0) + 1);
            }else if(!map.isEmpty()){
                int key = "1".equals(split[1]) ? map.lastKey() : map.firstKey();
                
                if(map.get(key) == 1) map.remove(key);
                else map.put(key, map.get(key) - 1);
            }
        }
        
        int[] dap = new int[2];
        
        if(!map.isEmpty()){
            dap[0] = map.lastKey();
            dap[1] = map.firstKey();
        }
        
        return dap;
    }
}
