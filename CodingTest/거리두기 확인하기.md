import java.util.*;
class Solution {
    public int[] solution(String[][] places) {
        char[][] arr = new char[5][5];
        int[][] di = {{0,1,0,-1},{1,0,-1,0}};
        int[] dap = new int[5];
        for(int i=0; i<places.length; i++){
            Queue<int[]> queue = new LinkedList<>();
            boolean[][] visited = new boolean[places.length][places[0].length];
            for(int j=0; j<places[0].length; j++){
                String s = places[i][j];
                for(int k=0; k<5; k++){
                    arr[j][k] = s.charAt(k);
                    if(s.charAt(k) == 'P'){
                        queue.add(new int[]{j,k,0,j,k});
                        visited[j][k] = true;
                    }
                }
            }
            
            boolean flag = true;
            while(!queue.isEmpty()){
                int[] cur = queue.poll();
                if(cur[2] > 2) continue;
                
                for(int a = 0; a<4; a++){
                    int dx = cur[0] + di[0][a];
                    int dy = cur[1] + di[1][a];
                    
                    if(dx >= 0 && dy >= 0 && dx < places.length && dy < places[0].length && arr[dx][dy] != 'X'){
                        if(arr[dx][dy] == 'P' && (dx != cur[3] || dy != cur[4])){
                            flag = false;
                            break;
                        }
                        
                        if(visited[dx][dy]) continue;
                        
                        queue.add(new int[]{dx,dy,cur[2] + 1, cur[3], cur[4]});
                        visited[dx][dy] = true;
                    }
                }
            }
            
            if(!flag){
                dap[i] = 0;
            }else{
                dap[i] = 1;
            }
        }
        
        return dap;
    }
}
