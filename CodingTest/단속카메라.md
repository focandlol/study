import java.util.*;
class Solution {
    public int solution(int[][] routes) {
        Arrays.sort(routes, (a, b) -> a[1] - b[1]);
        
        int end = -30001;
        int count = 0;
        
        for(int i=0; i<routes.length; i++){
            if(routes[i][0] > end){
                count++;
                end = routes[i][1];
            }
        }
        
        return count;
    }
}
