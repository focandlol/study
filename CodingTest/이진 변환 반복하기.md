class Solution {
    public int[] solution(String s) {
        int[] arr = new int[2];
        
        int zero = 0;
        int count = 0;
        
        while(!s.equals("1")){
            int size = 0;
            for(int i=0; i<s.length(); i++){
                if(s.charAt(i) == '1') size++;
            }
            
            zero += s.length() - size;
            
            String a = Integer.toBinaryString(size);

            count++;
            s = a;
        }
        
        arr[0] = count;
        arr[1] = zero;
        
        return arr;
    }
}
