class Solution {
    public String solution(String s) {
        String[] split = s.split("");
        
        StringBuilder sb = new StringBuilder();
        
        int idx = 0;
        for(String sp : split){
            idx = sp.equals(" ") ? 0 : idx + 1;
            sp = idx % 2 != 0 ? sp.toUpperCase() : sp.toLowerCase();
            sb.append(sp);
        }
        
        return sb.toString();
    }
}
