import java.util.*;

class Solution {
    public int solution(int bridge_length, int weight, int[] truck_weights) {
        ArrayDeque<Integer> bridge = new ArrayDeque<>();
        int count = 0;
        int we = 0;
        int idx = 0;

        for (int i = 0; i < bridge_length; i++) {
            bridge.add(0);
        }

        while (idx < truck_weights.length) {
            count++;

            we -= bridge.pollFirst();

            if (we + truck_weights[idx] <= weight) {
                bridge.addLast(truck_weights[idx]);
                we += truck_weights[idx];
                idx++;
            } else {
                bridge.addLast(0);
            }
        }

        return count + bridge_length;
    }
}
