class Solution {
    public int solution(String S) {
        int dap = 0;
        int count = 0;
        for(int i=0; i<S.length(); i++){
            dap += S.charAt(i) - '0';
            dap <<= 1;
        }
        dap >>= 1;

        while(dap > 0){
            if(dap % 2 == 0){
                dap /= 2;
            }else{
                dap -= 1;
            }
            count++;
        }
        return count;
    }
}
