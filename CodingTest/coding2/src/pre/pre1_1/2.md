import java.util.*;
import java.util.stream.Collectors;
class Solution {
    public int solution(String s) {
        int[] arr = new int[10];
        int max = 0;
        for(int i=0; i<s.length(); i++){
            arr[s.charAt(i) - '0']++;
            max = Math.max(max,arr[s.charAt(i) - '0']);
        }
        
        for(int i=0; i<arr.length; i++){
            if(arr[i] == max){
                return i;
            }
        }
        return 0;
}
}
