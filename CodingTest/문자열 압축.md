class Solution {
    public int solution(String s) {
        int a = s.length() / 2;
        int dap = s.length();
        
        for(int i=1; i<=a; i++){
            StringBuilder sb = new StringBuilder();
            int count = 1;
            String prev = s.substring(0,i);
            int idx = i;
            
            while(idx <= s.length()){
                int end = Math.min(idx + i, s.length());
                String st = s.substring(idx, end);
                if(prev.equals(st)){
                    count++;
                }else{
                    if(count != 1){
                        sb.append(count).append(prev);
                    }else{
                        sb.append(prev);
                    }
                    prev = st;
                    count = 1;
                }
                idx += i;
            }
            sb.append(prev);
            dap = Math.min(dap, sb.toString().length());
        }
        return dap;
    }
}
