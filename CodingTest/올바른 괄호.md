import java.util.*;
class Solution {
    boolean solution(String s) {
        ArrayDeque<Character> deque = new ArrayDeque<>();
        
        for(int i=0; i<s.length(); i++){
            char c = s.charAt(i);
            
            if(deque.isEmpty()) {
                deque.add(c);
                continue;
            }
            
            if(c == '(') deque.add(c);
            else if(deque.peekLast() == ')') return false;
            else deque.pollLast(); 
        }
        
        if(!deque.isEmpty()) return false;
        
        return true;
    }
}
