class Solution {
    public String solution(String s, int n) {
        
        char[] arr = s.toCharArray();
        StringBuilder sb = new StringBuilder();
        for(char c : arr){
            if(c == ' '){
                sb.append(" ");
                continue;
            }
            
            if(Character.isUpperCase(c)){
                c = (char)((c - 'A' + n) % 26 + 'A');
            }else{
                c = (char)((c - 'a' + n) % 26 + 'a');
            }
            
            sb.append(c);
        }
        return sb.toString();
    }
}
