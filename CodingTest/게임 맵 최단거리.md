import java.util.*;
class Solution {
    public int solution(int[][] maps) {
        int[][] di = {{0,1,0,-1},{1,0,-1,0}};
        
        Deque<int[]> deque = new ArrayDeque<>();
        boolean[][] visited = new boolean[maps.length][maps[0].length];
        
        deque.add(new int[]{0,0,1});
        visited[0][0] = true;
        
        while(!deque.isEmpty()){
            int[] cur = deque.poll();
            
            if(cur[0] == maps.length - 1 && cur[1] == maps[0].length - 1){
                return cur[2];
            }
            
            for(int i=0; i<4; i++){
                int dx = cur[0] + di[0][i];
                int dy = cur[1] + di[1][i];
                
                if(dx >= 0 && dx < maps.length && dy >= 0 && dy < maps[0].length && maps[dx][dy] != 0 && !visited[dx][dy]){
                    deque.add(new int[]{dx, dy, cur[2] + 1});
                    visited[dx][dy] = true;
                }
            }
        }
        return -1;
    }
}
