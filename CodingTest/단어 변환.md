import java.util.*;
class Solution {
    public int solution(String begin, String target, String[] words) {
        Queue<Word> queue = new LinkedList<>();
        boolean[] visited = new boolean[words.length];
        
        queue.add(new Word(begin, 0));
        
        while(!queue.isEmpty()){
            Word cur = queue.poll();
            
            if(cur.word.equals(target)) return cur.num;
            
            check(cur, visited, queue, words);
        }
        
        return 0;
    }
    
    void check(Word word, boolean[] visited, Queue<Word> queue, String[] words){
        for(int i=0; i<words.length; i++){
            if(visited[i]) continue;
            
            if(change(word.word, words[i])){
                queue.add(new Word(words[i], word.num + 1));
                visited[i] = true;
            }
        }
    }
    
    boolean change(String word1, String word2){
        int count = 0;
        for(int i=0; i<word1.length(); i++){
            if(word1.charAt(i) != word2.charAt(i)) count++;
            
            if(count >= 2) return false;
        }
        return true;
    }
    
    class Word {
        String word;
        int num;
        
        public Word(String word, int num){
            this.word = word;
            this.num = num;
        }
    }
}
