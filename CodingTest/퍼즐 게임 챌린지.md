class Solution {
    public int solution(int[] diffs, int[] times, long limit) {
        int left = 1;
        int right = 100000;
        
        while(left <= right){
            int mid = (left + right) / 2;
            if(check(diffs, times, limit, mid)) right = mid - 1;
            else left = mid + 1;
        }
        return left;
    }
    
    boolean check(int[] diffs, int[] times, long limit, int mid){
        long time = 0;
        for(int i=0; i<diffs.length; i++){
            if(mid >= diffs[i]) time += times[i];
            else{
                int prev = 0;
                if(i - 1 >= 0) prev = times[i-1];
                
                time += (diffs[i] - mid) * (times[i] + prev) + times[i];
            }
            
            if(time > limit) return false;
        }
        return true;
    }
}
