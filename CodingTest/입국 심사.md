class Solution {
    public long solution(int n, int[] times) {
        long start = 1;
        long end = (long) n * 1000000000;
        
        while(start <= end){
            long mid = (start + end) / 2;
            
            long sum = check(times, mid);
            
            if(sum < n){
                start = mid + 1;
            }else{
                end = mid - 1;
            }
        }
        return start;
    }
    
    public long check(int[] times, long mid){
        long sum = 0;
        for(int time : times){
            sum += mid / time;
        }
        return sum;
    }
}
