import java.util.*;
class Solution {
    Set<Set<String>> dap = new HashSet<>();
    public int solution(String[] user_id, String[] banned_id) {
        back(user_id, banned_id, 0, new HashSet<String>());
        return dap.size();
    }
    
    public void back(String[] user_id, String[] banned_id, int depth, Set<String> set){
        if(depth == banned_id.length){
            dap.add(set);
            return;
        }
        
        for(int i = 0; i < user_id.length; i++){
            if(set.contains(user_id[i])) continue;
            
            if(user_id[i].length() != banned_id[depth].length()) continue;
            
            if(check(user_id[i], banned_id[depth])) {
                set.add(user_id[i]);
                back(user_id, banned_id, depth + 1, new HashSet<String>(set));
                set.remove(user_id[i]);
            }
        }
    }
    
    public boolean check(String user_id, String banned_id){
        for(int i=0; i<user_id.length(); i++){
            if(banned_id.charAt(i) != '*' && user_id.charAt(i) != banned_id.charAt(i)) return false;
        }
        return true;
    }
}
