import java.util.*;
class Solution {
    public int solution(int n, int[][] computers) {
        ArrayList<Integer>[] graph = new ArrayList[n];
        
        for(int i=0; i<n; i++){
            graph[i] = new ArrayList<>();
        }
        
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(i == j) continue;
                
                if(computers[i][j] == 1){
                    graph[i].add(j);
                    graph[j].add(i);
                }
            }
        }
        
        boolean[] visited = new boolean[n];
        int count = 0;
        for(int i=0; i<n; i++){
            if(!visited[i]){
                dfs(graph, i, visited);
                count++;
            }
        }
        
        return count;
    }
    
    void dfs(ArrayList<Integer>[] graph, int a, boolean[] visited){
        
        visited[a] = true;
        
        for(int i : graph[a]){
            if(!visited[i]){
                dfs(graph, i, visited);
            }
        }
    }
}
