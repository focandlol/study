import java.util.*;
class Solution {
    Map<String, Integer> map = new HashMap<>();
    int score = 1;
    public int solution(String word) {
        String[] arr = {"A","E","I","O","U"};
        back(arr, 0 ,new StringBuilder());
        return map.get(word);
    }
    
    public void back(String[] arr, int depth, StringBuilder sb){
        if(depth == 5){
            return;
        }
        
        for(int i=0; i<arr.length; i++){
            StringBuilder s = new StringBuilder();
            s.append(sb).append(arr[i]);
            map.put(s.toString(), score++);
            back(arr,depth+1,s);
        }
    }
}
