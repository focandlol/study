class Solution {
    public int[] solution(long n) {
        StringBuilder sb = new StringBuilder();
        sb.append(n);
        sb.reverse();
        String s = sb.toString();
        
        int[] arr = new int[s.length()];
        for(int i=0; i<arr.length; i++){
            arr[i] = s.charAt(i) - '0';
        }
        return arr;
    }
}
