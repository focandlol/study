import java.util.*;
class Solution {
    int count = 0;
    public int solution(String s) {
        
        for(int i=0; i<s.length(); i++){
            check(s, i);
        }
        
        return count;
    }
    
    public void check(String s, int i){
        ArrayDeque<Character> deque = new ArrayDeque<>();
        
        for(int j=i; j<i+s.length(); j++){
            char c = s.charAt(j % s.length());
            
            if(c == '[' || c == '(' || c == '{') deque.addLast(c);
            else if(!deque.isEmpty() && (deque.peekLast() == c - 1 || deque.peekLast() == c - 2)) deque.pollLast();
            else return;
        }
        
        if(deque.isEmpty()) count++;
    }
}
