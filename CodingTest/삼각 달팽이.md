class Solution {
    public int[] solution(int n) {
        int[][] arr = new int[n][n];
        int[] dap = new int[n*(n+1)/2];
        
        int x = 0;
        int y = 0;
        int num = 1;
        int count = n;
        while(count > 0){
            for(int i=0; i<count; i++){
                arr[x][y] = num++;
                if(i != count - 1) x++;
            }
            count--;
            y++;
            
            for(int i=0; i<count; i++){
                arr[x][y] = num++;
                if(i != count -1) y++;
            }
            count--;
            x--;
            y--;
            
            for(int i=0; i<count; i++){
                arr[x][y] = num++;
                if(i != count -1){
                    x--;
                    y--;
                }
            }
            count--;
            x++;
        }
        
        int idx = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(arr[i][j] == 0) break;
                dap[idx++] = arr[i][j];
            }
        }
        
        return dap;
    }
}
