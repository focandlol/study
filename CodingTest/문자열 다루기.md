class Solution {
    public boolean solution(String s) {
        int len = s.length();
        
        if(len != 4 && len != 6) return false;
        
        for(int i=0; i<s.length(); i++){
            if(!Character.isDigit(s.charAt(i))) return false;
        }
        
        return true;
    }
}
