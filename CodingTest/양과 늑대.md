import java.util.*;
class Solution {
    int max = 0;
    public int solution(int[] info, int[][] edges) {
        ArrayList<Integer>[] graph = new ArrayList[info.length];
        
        for(int i=0; i<graph.length; i++){
            graph[i] = new ArrayList<>();
        }
        
        for(int[] arr : edges){
            graph[arr[0]].add(arr[1]);
        }
        
        ArrayList<Integer> vi = new ArrayList<>();
        vi.add(0);
        
        find(0, 0, 0, graph, vi, info);
        
        return max;
    }
    
    void find(int node, int s, int w, ArrayList<Integer>[] graph, ArrayList<Integer> vi, int[] info){
        if(info[node] == 0) s++;
        else w++;
        
        if(s <= w) return;
        
        max = Math.max(max, s);
        
        vi.remove(Integer.valueOf(node));
        
        for(int n : graph[node]){
            vi.add(n);
        }
        
        for(int v : vi){
            find(v, s, w, graph, new ArrayList<Integer>(vi), info);
        }
    }
}
