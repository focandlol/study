import java.util.*;
class Solution {
    static ArrayList<int[]> list = new ArrayList<>();
    public int[][] solution(int n) {
        find(n,1,3,2);
        
        int[][] arr = new int[list.size()][2];
        
        for(int i=0; i<list.size(); i++){
            int[] ar = list.get(i);
            
            arr[i][0] = ar[0];
            arr[i][1] = ar[1];
        }
        
        return arr;
    }
    
    public void find(int n, int from, int to, int sub){
        int[] arr = {from, to};
        if(n == 1){
            list.add(arr);
            return;
        }
        
        find(n-1, from, sub, to);
        list.add(arr);
        find(n-1, sub, to, from);
    }
}
