class Solution {
    public int[] solution(int brown, int yellow) {
        int[] dap = new int[2];
        
        for(int i=(int) Math.ceil(Math.sqrt(yellow)); i<=yellow; i++){
            if(yellow % i != 0) continue;
            
            int garo = i;
            int sero = yellow / i;
            
            if((garo + 2) * (sero + 2) - yellow == brown){
                dap[0] = garo + 2;
                dap[1] = sero + 2;
                break;
            }
        }
        return dap;
    }
}
